var a = 1, b = 1, c, d;
c = ++a; alert(c); // 2 сначала инкрементирование, потом присваивание значения
d = b++; alert(d); // 1 идет сначала присваивание, а потом увеличение на 1
c = (2+ ++a); alert(c); // 5 к двойке прибавляется переменная а, которая была равна 2, увеличенная на 1, соотвественно 2+3=5
d = (2+ b++); alert(d); // 4 к двойке прибавляется переменная b, равная 2, а увеличение переменной b идет после операции сложения 
alert(a); // 3 произошло увеличение на 1, а потом вывод переменной
alert(b); // 3 произошел вывод в предыдущей операции, а потом увеличение на 1, как результат - значение 3.
